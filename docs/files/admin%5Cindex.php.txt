<?php 
if ( ! defined('BASEPATH')) exit('No direct script access allowed');

// Base admin class

class Index extends Admin_Controller 
{

	public function __construct()
	{
		parent::__construct();
	}

	public function index()
	{		
		$data = array(
			'title' => "Главная",
			'meta_title' => "Главная",
			'error' => "",
			'user' => $this->user,
			'menu' => $this->menu,
			'url' => "/".$this->uri->uri_string()
		);

		$this->load->view('admin/admin.php', $data);
	}
}
